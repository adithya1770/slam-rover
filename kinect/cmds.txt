to install depn.

sudo apt update
sudo apt install git cmake build-essential libusb-1.0-0-dev python3-pip -y
pip3 install numpy opencv-python


clone and build libfreenect

git clone https://github.com/OpenKinect/libfreenect.git
cd libfreenect
mkdir build && cd build

cmake .. -DBUILD_PYTHON3=ON
make
sudo make install
sudo ldconfig

test the kinect

lsusb
